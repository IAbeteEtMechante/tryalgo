[docs]def readarray(typ):
    """
    function to read an array
    """
    return list(map(typ, stdin.readline().split()))

[docs]def readmatrix(n):
    """
    function to read a matrix
    """
    M = []
    for _ in range(n):
        row = readarray(int)
        assert len(row) == n
        M.append(row)
    return M
